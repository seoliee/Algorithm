int lastIndex 선언

// size
int size = 차수 * 차수

// 생성자
함수 Tree(int size)
	tree를 크기가 size인 배열로 초기화
	tree 배열의 모든 요소를 0으로 채우기
	lastIndex = -1으로 설정 (왜냐하면 지금 트리에 노드가 없으므로)
	
	
// 노드 추가
함수 addNode(String data)
	lastIndex + 1이 size보다 작은 경우
		lastIndex + 1
		tree[lastIndex]에 data값 저장
	그렇지 않은 경우
		"저장 못함여" 출력
	조건문 종료

// 부모 노드 반환
함수 getParentIdx(int idx)
	idx > 0이고, idx <= lastIndex인 경우
		(idx - 1) / 2 반환
	그렇지 않은 경우
		-1 반환
	조건문 종료
	
// 왼쪽 자식 노드 반환
함수 getLeftChildIdx(int idx)
	int leftChildIdx = (idx * 2) + 1
	만약 leftChildIdx <= lastIndex라면
		leftChildIdx 반환
	그렇지 않은 경우
		-1 반환
	조건문 종료

// 오른쪽 자식 노드 반환
함수 getRightChildIdx(int idx)
	int rightChildIdx = (idx * 2) + 2
	만약 rightChildIdx <= lastIndex라면
		rightChildIdx 반환
	그렇지 않은 경우
		-1 반환
	조건문 종료
